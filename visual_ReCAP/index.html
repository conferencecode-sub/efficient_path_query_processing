WITH RECURSIVE Paths AS (
  SELECT
    start_id AS v,
    0 AS depth,
    'q0' AS q,
    init() AS d
  UNION ALL
  SELECT
    e.dst AS v,
    p.depth + 1 AS depth,
    t.next_q AS q,
    update_d(p.d, p.q, t.next_q, edge_struct(e)) AS d
  FROM Paths p
  JOIN nfa_transitions t ON p.q = t.q
  JOIN edges e ON e.src = p.v AND e.label = t.label
  WHERE p.depth <= $upper_bound
    AND is_viable_d(p.d, p.q, t.next_q, edge_struct(e))
)
SELECT *
FROM Paths
WHERE depth BETWEEN $lower_bound AND $upper_bound
  AND q IN ('qF')
  AND is_viable_d_final(finalize(d));
